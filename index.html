jQueryMobile - PopupWrapper

A Popup Dialog Box for jQueryMobile 1.3.0+

PopupWrapper aims to be a wrapper for the built in popups in jQM - with an emphasis on single-use dynamically generated information.

$('#blnkmode').on('vclick', function() {
  $('<div>').popupwrapper({
    displayMode: 'blank',
    closeButton: 'right',
    content:
      "<div data-role='header'><h1>Antigonish</h1></div>" +
      "<div data-role='content'><p>Yesterday, upon the stair,<br />..</p></div>"
  });
});
$('#butbutmode').on('vclick', function() {
  $('<div>').popupwrapper({
    displayMode: 'button',
    buttonMode:  'button',
    closeButton: 'right',
    headerText:  'Buttons',
    subTitle:    'Pick one, eh?',
    headerMinWidth: '300px',
    buttons: {
      'OK': {
        click: function () { alert('You clicked OK'); }
      },
      'Cancel': {
        click: function () { alert('You clicked CANCEL'); },
        icon: "delete",
        theme: "c",
      }
    }
  });
});
